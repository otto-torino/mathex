Example
 
 var question = new mathex.Question({
  text: '2. Which of the following is an identity?',
  answers: [
    '{% 3x = 9 %}',
    '{% 2a = 5 %}',
    '{% 7 + 8 = 16 %}',
    '{% 15 - 9 = 6 %}'
  ],

})
